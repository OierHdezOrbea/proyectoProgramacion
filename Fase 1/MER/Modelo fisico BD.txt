CREATE TABLE Centro_de_trabajo(
CENTRO_NO INTEGER NOT NULL,
NOMBRE VARCHAR2(20) NOT NULL,
CALLE VARCHAR2 (20) NOT NULL,
TELEFONO CHAR(9) NOT NULL,
COD_POSTAL CHAR(5) NOT NULL,
CIUDAD VARCHAR2(15) NOT NULL,
PROVINCIA VARCHAR2(15) NOT NULL,
PRIMARY KEY (CENTRO_NO)
);

CREATE TABLE Trabajador(
DNI INTEGER NOT NULL,
FECHA_NAC DATE,
SALARIO NUMBER(8,2),
NOMBRE VARCHAR2(15) NOT NULL,
APELLIDO1 VARCHAR2(15) NOT NULL,
APELLIDO2 VARCHAR2(15) NOT NULL,
DIRECC VARCHAR2(50) NOT NULL,
TELEFONO_PER CHAR(9),
TELEFONO_EMP CHAR(9) NOT NULL,
CENTRO_NO INTEGER NOT NULL,
AVISOS VARCHAR2(200),
CATEGORIA VARCHAR2(20) NOT NULL,
PRIMARY KEY (DNI),
CONSTRAINT Centro_Trabajador_FK FOREIGN KEY (CENTRO_NO) REFERENCES Centro_de_trabajo (CENTRO_NO)
);

CREATE TABLE Usuario(
USU_NO INTEGER NOT NULL,
CONTRASENA VARCHAR2(8) NOT NULL,
DNI CHAR(9) NOT NULL,
PRIMARY KEY (USU_NO),
CONSTRAINT Usuario_Trabajador_FK FOREIGN KEY (DNI) REFERENCES Trabajador (DNI)
);

CREATE TABLE Vehiculo(
ID_VEHICULO INTEGER NOT NULL,
MARCA VARCHAR2(15) NOT NULL,
MODELO VARCHAR2(15) NOT NULL,
COLOR VARCHAR2(15),
DNI CHAR(9) NOT NULL,
PRIMARY KEY (ID_VEHICULO),
CONSTRAINT Vehiculo_Trabajador_FK FOREIGN KEY (DNI) REFERENCES Trabajador (DNI)
);

CREATE TABLE Parte(
PARTE_COD INTEGER,
HORA_INIC DATE,
HORA_FIN DATE,
ESTADO BOOLEAN,
DNI CHAR(9),
PRIMARY KEY (PARTE_COD),
CONSTRAINT Parte_Trabajador_FK FOREIGN KEY (DNI) REFERENCES Trabajador (DNI)
);

CREATE TABLE Albaran(
ALB_NO INTEGER,
KM_INICIO INTEGER,
KM_FIN INTEGER,
GASTO_COMBUSTIBLE NUMBER(6,2),
GASTO_AUTOPISTA NUMBER(6,2),
GASTO_DIETAS NUMBER(6,2),
OTROS_GASTOS NUMBER(6,2),
INCIDENCIAS VARCHAR2(200),
PRIMARY KEY (ALB_NO),
CONSTRAINT Albaran_Parte_FK FOREIGN KEY (PARTE_COD) REFERENCES Parte (PARTE_COD)
);
